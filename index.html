function tribonacci(signature,n){
  if (n == 0) return []
  if(n <= 3){
    signature.length = n
    return signature
  }

  while (n > 3){
    let l = signature.length
    signature.push(signature[l-1] + signature[l-2] + signature[l-3])
    n--
  }
  return signature
}
